@startuml

Title Диаграмма последовательности
autonumber

actor Пользователь as U
participant Frontend as F
participant Backend as B
database БД as DB

U -> F   : Открывает главную страницу
F -> B   : Берем данные для главной
B -> DB  : Запрос в БД 
B <-- DB : Выборка из БД для главной
F <-- B  : Данные для отображения главной
U <-- F  : Главная страница с разделами

U -> F   : Выбирает раздел
F -> B   : Запрашиваем данные по параметру
B -> DB  : Запрос в БД по параметру
B <-- DB : Выборка из БД по разделу
F <-- B  : Данные для отображения в разделе
U <-- F  : Раздел с фильмами

U -> F   : Выбирает фильм
F -> B   : Запрашиваем данные по фильму
B -> DB  : Запрос в БД по параметру
B <-- DB : Выборка из БД по фильму
F <-- B  : Данные для отображения инфомрации о фильме
U <-- F  : Информация о фильме

@enduml
